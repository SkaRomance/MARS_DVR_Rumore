import pathlib
import pytest

async def mars_mock_server():
    """Minimal mock MARS HTTP server for E2E."""
    app = web.Application()

    dvr_data = json.loads((pathlib.Path(__file__).parent / "fixtures" / "dvr_mock.json").read_text())

    async def me(request):
        return web.json_response({
            "userId": "mock-user-id",
            "email": "test@example.com",
            "tenantId": dvr_data["tenantId"],
            "tenants": [{"id": dvr_data["tenantId"], "name": "ACME", "role": "consultant"}],
            "enabledModules": ["dvr", "noise"],
            "rolesByCompany": {},
        })

    async def verify(request):
        return web.json_response({"enabled": True, "moduleKey": "noise"})

    async def get_revision(request):
        return web.json_response(dvr_data)

    async def put_module_ext(request):
        body = await request.json()
        # Increment version and echo
        dvr_data["version"] += 1
        if "module_extensions" not in dvr_data["snapshot"]:
            dvr_data["snapshot"]["module_extensions"] = {}
        dvr_data["snapshot"]["module_extensions"]["noise"] = body
        return web.json_response({"version": dvr_data["version"], "updatedAt": "2026-04-17T10:00:00Z"})

    app.router.add_get("/auth/me", me)
    app.router.add_post("/client-app/compliance/modules/verify/{key}", verify)
    app.router.add_get("/dvr-documents/{d}/revisions/{r}", get_revision)
    app.router.add_put("/dvr-documents/{d}/revisions/{r}/module-extensions/{key}", put_module_ext)

    runner = web.AppRunner(app)
    await runner.setup()
    site = web.TCPSite(runner, "localhost", 5999)
    await site.start()

    yield "http://localhost:5999"

    await runner.cleanup()

"""E2E browser test: autopilot flow end-to-end."""
import pytest
from playwright.async_api import async_playwright, expect


@pytest.mark.e2e
@pytest.mark.asyncio
async def test_consultant_runs_autopilot_and_approves(mars_mock_server, valid_jwt, app_running):
    """
    Full scenario:
    1. Open Rumore UI as iframe with ?token=&dvr_doc_id=&revision_id=
    2. Click "Avvia Valutazione AI"
    3. Observe progress events (at least 5 step_started + 5 step_completed)
    4. Result card appears with LEX value
    5. Click "Approva tutto"
    6. Verify suggestion list shows approved
    7. Verify MARS mock received PUT on module_extensions.noise
    """
    async with async_playwright() as pw:
        browser = await pw.chromium.launch()
        page = await browser.new_page()

        url = f"http://localhost:8085/static/index.html?token={valid_jwt}&dvr_doc_id=mock-doc-id&revision_id=mock-revision-id&mars_api={mars_mock_server}"

        # Listen for console errors
        errors = []
        page.on("console", lambda msg: errors.append(msg.text) if msg.type == "error" else None)

        await page.goto(url)

        # Wait for autopilot view render
        await expect(page.locator("#btn-start")).to_be_visible(timeout=10000)

        # Start autopilot
        await page.click("#btn-start")

        # Wait for completion (result card visible)
        await expect(page.locator(".result-card")).to_be_visible(timeout=120000)  # 2 min max

        # Verify LEX shown
        lex_text = await page.locator(".stat-value").first.text_content()
        assert "dB" in lex_text or any(c.isdigit() for c in lex_text)

        # Approve all
        await page.click("#btn-approve-all")

        # Wait for toast
        await expect(page.locator(".toast-success, .toast")).to_be_visible(timeout=5000)

        # No JS errors
        assert not errors, f"Console errors: {errors}"

        await browser.close()
```

"""Locust load test for autopilot endpoint."""
from locust import HttpUser, task, between
import jwt
import time


class ConsultantUser(HttpUser):
    wait_time = between(2, 5)
    host = "http://localhost:8085"

    def on_start(self):
        self.token = jwt.encode(
            {
                "sub": "load-test-user",
                "email": "load@test.com",
                "tenant_id": "load-test-tenant",
                "type": "access",
                "exp": int(time.time()) + 3600,
            },
            "test-secret",
            algorithm="HS256",
        )
        self.headers = {"Authorization": f"Bearer {self.token}"}

    @task(5)
    def poll_status(self):
        self.client.get("/api/v1/noise/autopilot/some-id/status", headers=self.headers)

    @task(3)
    def list_suggestions(self):
        self.client.get("/api/v1/noise/suggestions/by-context/some-id", headers=self.headers)

    @task(1)
    def bootstrap_context(self):
        self.client.post(
            "/api/v1/noise/contexts/bootstrap",
            json={"mars_dvr_document_id": "doc-1", "mars_revision_id": "rev-1"},
            headers=self.headers,
        )
```
